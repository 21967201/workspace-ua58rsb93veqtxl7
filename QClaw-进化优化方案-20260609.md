```python
        bottlenecks = self.identify_bottlenecks(performance_data)
        
        # Step 3: 策略变异
        strategies = []
        strategies += self.rule_evolution(patterns)      # AGENTS.md规则优化
        strategies += self.param_tuning(bottlenecks)     # 参数自动调优
        strategies += self.tech_integration(tech_breakthroughs)  # 新技术接入
        
        # Step 4: 评估选择
        for strategy in strategies:
            strategy.score = self.simulate(strategy)
        best = max(strategies, key=lambda s: s.score)
        
        # Step 5: 应用并验证
        self.apply(best)
        self.verify_baseline()
```
